ใช้ Google Apps Script (ฟรี) เพื่อเก็บข้อมูลบน Google Sheet — sync ได้ทุกอุปกรณ์
🔑 ส่วนที่ 0 — ตั้งค่า Google Apps Script (sync ข้ามอุปกรณ์)
1
เปิด Google Apps Script
ไปที่ script.google.com แล้ว Login ด้วย Google Account
2
สร้างโปรเจกต์ใหม่
กด "New Project" → ตั้งชื่อ เช่น PalmDonsak
3
วางโค้ดนี้ลงในช่อง Code.gs (แทนที่โค้ดเดิมทั้งหมด)
const SPREADSHEET_ID = '1Y3SJzywuCzjZEn9OwjKK3WkZzMvRpCzKq_gGt8ABrrc'; const SHEET_NAME = 'PalmData'; function getSheet() { var ss = SpreadsheetApp.openById(SPREADSHEET_ID); return ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); } function doGet(e) { try { var action = e.parameter.action; if (action === 'load') { var sh = getSheet(); var data = sh.getRange(1,1).getValue(); var items = data ? JSON.parse(data) : []; return ContentService.createTextOutput(JSON.stringify({items: items})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:false,msg:'unknown action'})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var body = JSON.parse(e.postData.contents); if (body.action === 'save') { var sh = getSheet(); sh.getRange(1,1).setValue(JSON.stringify(body.items)); // บันทึกแบบ human-readable ใน Sheet2 ด้วย writeReadableSheet(body.items); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:false})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function writeReadableSheet(items) { try { var ss = SpreadsheetApp.openById(SPREADSHEET_ID); var sh2 = ss.getSheetByName('รายการ') || ss.insertSheet('รายการ'); sh2.clearContents(); var headers = ['ID','วันที่','ประเภท','หมวดหมู่','รายการ','น้ำหนัก','ราคา/กก.','รับรวม','ค่าแรง','สุทธิ','หมายเหตุ']; sh2.appendRow(headers); items.forEach(function(it) { sh2.appendRow([it.id, it.date, it.type, it.category, it.desc, it.weight||0, it.price||0, it.grossAmount||0, it.labor||0, it.amount||0, it.note||'']); }); } catch(e) { /* ไม่ต้อง throw */ } }
4
📋 Google Sheet ที่ใช้งาน
ระบบเชื่อมกับ Spreadsheet นี้โดยตรง:

📊 เปิด Google Sheet →

ข้อมูลจะถูกบันทึกใน 2 แท็บ:
• PalmData — JSON สำหรับระบบ (อย่าแก้ด้วยมือ)
• รายการ — ตารางอ่านง่ายสำหรับดูข้อมูล
5
Deploy เป็น Web App
กด Deploy → New Deployment
→ Type: Web app
→ Execute as: Me
→ Who has access: Anyone
→ กด Deploy → คัดลอก Web app URL
6
ใส่ URL ในไฟล์ index.html
เปิดไฟล์ index.html ด้วย Notepad / VSCode แล้วหาบรรทัด:
var GAS_URL = 'https://script.google.com/macros/s/AKfycbzpl0pgmFOK-bJDa4xAvJmdKvT2Nqs6TaSbz6_MHHVSrU-tj6KS8UBNm5uJQee-FR6h/exec';
แก้เป็น URL จริงที่ได้จาก Deploy:
var GAS_URL = 'https://script.google.com/macros/s/YOUR_ID/exec';
⚠️ แทน YOUR_ID ด้วย ID จริงที่ได้จาก Deploy แล้วบันทึกไฟล์
✅
เสร็จแล้ว!
ข้อมูลจะ sync อัตโนมัติทุกครั้งที่เพิ่ม/แก้ไข/ลบรายการ และโหลดข้อมูลจาก Cloud เมื่อเปิดจากทุกอุปกรณ์
📌 ส่วนที่ 1 — สมัครสมาชิก GitHub
1
เปิด github.com
ไปที่ github.com บนเบราวเซอร์คอมหรือมือถือ
2
กด "Sign up"
กรอก อีเมล → รหัสผ่าน → ชื่อผู้ใช้ (username) → ยืนยันอีเมล
⚠️ จด username ไว้ เพราะจะอยู่ใน URL เว็บของคุณ
3
Login เข้าระบบ
กด Sign in → ใส่อีเมล + รหัสผ่าน
📁 ส่วนที่ 2 — สร้าง Repository
1
กดปุ่ม "+" แล้วเลือก "New repository"
อยู่มุมบนขวาหลัง Login
2
ตั้งชื่อ Repository
พิมพ์ในช่อง "Repository name":
palm-donsak
ใช้ตัวเล็กและ - เท่านั้น ห้ามมีช่องว่าง
3
เลือก "Public"
คลิก radio button "Public" (ต้องเลือกนี้จึงจะใช้ Pages ฟรีได้)
4
กด "Create repository" สีเขียว
เลื่อนลงด้านล่างแล้วกด รอสักครู่
📤 ส่วนที่ 3 — อัพโหลดไฟล์ index.html
1
บันทึกไฟล์เว็บนี้เป็น index.html
กดดาวน์โหลดจากระบบ หรือกด Ctrl+S แล้วบันทึกเป็น index.html
2
ในหน้า Repository กด "Add file" → "Upload files"
ปุ่มนี้อยู่กลางหน้าหลัง create repository
3
ลากไฟล์ index.html วางลง หรือกด "choose your files"
เลือกไฟล์ index.html ที่บันทึกไว้
4
กด "Commit changes" สีเขียว
เลื่อนลงด้านล่างแล้วกด รอระบบอัพโหลด
⚙️ ส่วนที่ 4 — เปิดใช้งาน GitHub Pages
1
กด "Settings" ในแถบเมนูบนของ Repository
2
เลือก "Pages" จากเมนูซ้าย
อยู่ในกลุ่ม "Code and automation"
3
ตั้งค่า Source
Source → Deploy from a branch
Branch → เลือก main → เลือก / (root)
4
กด "Save" แล้วรอ 1–3 นาที
✅
เว็บพร้อมใช้งาน!
URL ของคุณ:
https://USERNAME.github.io/palm-donsak
แทน USERNAME ด้วยชื่อที่สมัครไว้
📱 ติดตั้งบนมือถือ (เหมือนแอพ)
🍎 iPhone / iOS
1. เปิด URL ใน Safari
2. กดปุ่มแชร์ 📤
3. เลือก "เพิ่มลงในหน้าจอโฮม"
4. กด เพิ่ม
🤖 Android
1. เปิด URL ใน Chrome
2. กดปุ่ม ⋮ เมนู
3. เลือก "เพิ่มในหน้าจอหลัก"
4. กด เพิ่ม
